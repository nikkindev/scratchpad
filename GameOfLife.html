# Find number of active neighbours
function sum_neighbours(generation,i,j)
    neighbours = 0
    if checkbounds(Bool,generation,i-1,j-1)
        neighbours += generation[i-1,j-1]
    end
    if checkbounds(Bool,generation,i-1,j)
        neighbours += generation[i-1,j]
    end
    if checkbounds(Bool,generation,i-1,j+1)
        neighbours += generation[i-1,j+1]
    end
    if checkbounds(Bool,generation,i,j-1)
        neighbours += generation[i,j-1]
    end
    if checkbounds(Bool,generation,i,j+1)
        neighbours += generation[i,j+1]
    end
    if checkbounds(Bool,generation,i+1,j-1)
        neighbours += generation[i+1,j-1]
    end
    if checkbounds(Bool,generation,i+1,j)
        neighbours += generation[i+1,j]
    end
    if checkbounds(Bool,generation,i+1,j+1)
        neighbours += generation[i+1,j+1]
    end
    return neighbours
end


# If a dead cell has exactly 3 active neighbours, it gets activated
# If an alive cell has less than 2 or more than 4 active neighbours, it dies
function next_generation(generation)
    # Check for neighbours with a copy of the current generation and 
    # update for the next generation in the original matrix
    current_generation = copy(generation) 
    l,b = size(generation)
    for i in 1:l, j in 1:b
        neighbours = sum_neighbours(current_generation,i,j)
        if current_generation[i,j] == 0 # Dead cell
            if neighbours == 3
                generation[i,j] = 1 # Comes to life for 3 neighbours
            end
        else # Cell is alive
            if neighbours < 2 || neighbours > 3
                generation[i,j] = 0 # Cell dies
            end
        end
    end
    return generation
end

l, b = 100, 200 # Length and breadth
generation = zeros(Int8,l,b) # Initialize generation
p = 0.1 # Probability that a cell becomes active
for i in 1:l, j in 1:b
     generation[i,j] = rand() < p ? 1 : 0
end
